    x_hat: \bar{x} = \frac{1}{n}\sum_{i=1}^{n}x_{i}
    y_hat: \bar{y} = \frac{1}{n}\sum_{i=1}^{n}y_{i}
    weight: W = \frac{\sum_{i=1}^{n}(x_{i}-\bar{x})y_{i}}{\sum_{i=1}^{n}(x_{i}-\bar{x})x_{i}}
    bias: b = \bar{y} - W\bar{x}
    lineFit: y = b + Wx
